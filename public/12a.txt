#!/bin/bash
# Program to find the largest of the given array of numbers
declare -A arr
echo "Enter three elements of the array: "
for ((i=0; i<3; i++))
do
    read arr["$i"]
done
max=${arr[0]}
for ((i=1; i<3; i++))
do
    if [[ ${arr[$i]} -gt $max ]]
    then
        max=${arr[$i]}
    fi
done

echo "The largest of the given numbers is $max"
